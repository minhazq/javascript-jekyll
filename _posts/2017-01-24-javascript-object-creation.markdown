There is 3 ways you can create an Object.
Literal Form j:

{% highlight ruby %}
def print_hi(name)
  puts "Hi, #{name}"
end
print_hi('Tom')
#=> prints 'Hi, Tom' to STDOUT.
{% endhighlight %}

{% highlight java%}
"use strict"

var Emp = require("./Employee");

var emp1 = Emp.getInstance();
var emp2 = Emp.getInstance();

console.log(emp1.getNum());
console.log(emp2.getNum());
{% endhighlight %}
