There is 3 ways you can create an Object.
Literal Form :

{% highlight ruby %}
def print_hi(name)
puts "Hi, #{name}"
end
print_hi('Tom')
\#=> prints 'Hi, Tom' to STDOUT.
{% endhighlight %}

{% highlight js %}
var test = function(){};
{% endhighlight %}

{% highlight java %}
String st = new String("test");

System.out.println(st);
{% endhighlight %}
